6class ResBlock3d(nn.Module): 

7 """  

8 Based on 

9 https://towardsdev.com/implement-resnet-with-pytorch-a9fb40a77448  

10 https://github.com/pytorch/vision/blob/main/torchvision/models/resnet.py 

11 """ 

12 def __init__(self, in_channels:int, out_channels:int, downsample:bool): 

13 super().__init__() 

14 if downsample: 

15 self.conv1 = nn.Conv3d(in_channels, out_channels, kernel_size=3, stride=2, padding=1) 

16 self.shortcut = nn.Sequential( 

17 nn.Conv3d(in_channels, out_channels, kernel_size=1, stride=2), 

18 nn.BatchNorm3d(out_channels) 

19 ) 

20 else: 

21 self.conv1 = nn.Conv3d(in_channels, out_channels, kernel_size=3, stride=1, padding=1) 

22 self.shortcut = nn.Sequential() 

23 

24 self.conv2 = nn.Conv3d(out_channels, out_channels, kernel_size=3, stride=1, padding=1) 

25 self.bn1 = nn.BatchNorm3d(out_channels) 

26 self.bn2 = nn.BatchNorm3d(out_channels) 

27 self.relu = nn.ReLU(inplace=True) 

28 

29 def forward(self, x): 

30 shortcut = self.shortcut(x) 

31 x = self.relu(self.bn1(self.conv1(x))) 

32 x = self.relu(self.bn2(self.conv2(x))) 

33 x = x + shortcut 

34 return self.relu(x)

37class DownBlock(nn.Module): 

38 def __init__( 

39 self, 

40 in_channels:int = 1, 

41 downsample:bool = True, 

42 ): 

43 super().__init__() 

44 self.in_channels = in_channels 

45 self.out_channels = in_channels 

46 if downsample: 

47 self.out_channels *= 2

51

59 def __init__( 

60 self, 

61 in_channels:int = 1, 

62 initial_features:int = 64,

67 self.in_channels = in_channels

92class ResNet3d(nn.Module): 

93 def __init__( 

94 self, 

95 num_classes:int=1, 

96 body = None, 

97 in_channels:int = 1, 

98 initial_features:int = 64, 

99 ): 

100 super().__init__() 

101 self.body = body if body is not None else ResNet3dBody(in_channels=in_channels, initial_features=initial_features) 

102 assert in_channels == self.body.in_channels 

103 assert initial_features == self.body.initial_features 

104 self.global_average_pool = torch.nn.AdaptiveAvgPool3d(1) 

105 self.final_layer = torch.nn.Linear(self.body.output_features, num_classes) 

106 

107 def forward(self, x: Tensor) -> Tensor: 

108 x = self.body(x) 

109 # Final layer 

110 x = self.global_average_pool(x) 

111 x = torch.flatten(x, 1) 

112 output = self.final_layer(x) 

113 return output
